DECISION DE DISEÑO 1: Para saber si una asignacion esta bien tipada almacenamos
                      la variable y su tipo al procesar la declaracion de variables.

(almacen variables)         variable    | tipo
                       ----------------------------------
                             i,o        | NUM
                             mx         | SEQ(NUM)
                             s          | LOG
                             d          | SEQ(LOG)

programa : PROGRAMA f=variables{almacenar f en almacen de variables} subprogramas instrucciones EOF

(parametro de salida f)
variables : VARIABLES d=lista_variables* {almacenar d en f};

(parametro de salida d)
lista_variables : IDENT{incluir IDENT en v} COMA s=lista_variables{incluir s en v}
    | IDENT{incluir IDENT en v} DP t=tipo PyC {almacenar en d cada v con tipo t}
    ;

(parametro de salida t)
tipo :  t=SEQ PA NUM PC
        |t=SEQ PA LOG PC
        |t=NUM
        |t=LOG
        ;

DECISION DE DISEÑO 2: Calculo del tipo de una expresion

----------------------------------------
(funcion tipo_op_aritmetica(t1, t2)
                si t1 o t2 igual no tipo entonces
                    tipo = no tipo
                sino
                    si t1 y t2 igual NUM entonces
                        tipo = NUM
                    sino
                        tipo = no tipo)
-----------------------------------------
(funcion tipo_op_booleana(t1,t2)
                si t1 o t2 igual no tipo entonces
                    tipo = no tipo
                sino
                    si t1 y t2 igual LOG entonces
                        tipo = LOG
                    sino
                        tipo = no tipo)
-----------------------------------------
(funcion tipo_seq_elems(tipos)
                si elemento en tipos igual a no tipo
                    tipo = no tipo
                sino
                    si todos elementos en tipos igual a NUM entonces
                        tipo = SEQ PA NUM PC
                    si todos elementos en tipos igual a LOG entonces
                        tipo = SEQ PA LOG PC
                    sino
                        tipo = no tipo)
